require("framework/platform/AndroidGamePlatform")
require("framework/common/kakura/MethodCommon")
require("framework/manager/LogsManager")

IosGamePlatform = class('IosGamePlatform',AndroidGamePlatform)
function IosGamePlatform:ctor()
  IosGamePlatform.super.ctor(self)
end

function IosGamePlatform:getGlobalMethod()
  return MethodCommon.global_Account_loginIosMaster
end

function IosGamePlatform:checkSignSha1()
  if not self._nativeBackData or not self._nativeBackData.boundleId then
    LogsManager.warn("SignSha1 native boundleId数据还没初始化")
    return
  end
  local rightId = self:getPackageName()
  local len = #rightId
  local targetId = self._nativeBackData.boundleId
  if targetId:slice(-len) ~= rightId then
    LogsManager.errorTag("boundleIdError", "包名不符合.目标包名:" .. targetId)
    self.childChannelId = targetId
  else
    LogsManager.echo("这个包是合法的")
  end
end
