BannerOrRecommendServer = class('BannerOrRecommendServer')
function BannerOrRecommendServer.BannerOrRecommendSend(callBack, thisObj)

    callBack=nil
  end
  if thisObj == nil then
    thisObj=nil
  end
  local upData = {}
  -- [ts2lua]upData下标访问可能不正确

  local backData = Client.instance:doDummyServerBack(nil, upData, nil)
  SingleCommonServer.startSaveClientData()
  callBack and callBack:call(thisObj)
end
