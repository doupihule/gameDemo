require("game/sys/model/BaseModel")
require("framework/common/kakura/Client")

JumpCommonModel = class('JumpCommonModel',BaseModel)
function JumpCommonModel:ctor()
  JumpCommonModel.super.ctor(self)
end

function JumpCommonModel.__getters.instance()
  if not JumpCommonModel._instance then
    JumpCommonModel._instance = JumpCommonModel()
  end
  return JumpCommonModel._instance
end

function JumpCommonModel:initData(d)
  JumpCommonModel.super.initData(self,d)
end

function JumpCommonModel:updateData(d)
  JumpCommonModel.super.updateData(self,d)
end

function JumpCommonModel:getJumpedList()
  local jumpList = self._data.jumpedList or {}
  if not jumpList.expireTime or Client.instance.serverTime > jumpList.expireTime then
    self._data.jumpedList = {}
    return {}
  else
    return jumpList.appList
  end
end
