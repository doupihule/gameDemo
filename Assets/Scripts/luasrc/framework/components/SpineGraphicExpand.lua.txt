-- 2dspine 对象
SpineGraphicExpand = class("SpineGraphicExpand",LuaBaseView)

--spine动画对象
SpineGraphicExpand.__spineAni = nil
--构造器函数
function SpineGraphicExpand:ctor(name, boundlePath)
    SpineGraphicExpand.super.ctor(self)
    if(not boundlePath) then boundlePath = name end
    self:setCObject(ResourceManager.loadSpinePrefab(name,boundlePath))
    self.__spineAni = self.__cobject:GetComponent("SkeletonGraphic")
end

--播放动画
function SpineGraphicExpand:play(label, loop)
    self.__spineAni.freeze = false
    print(label,loop,"__SpineGraphicExpand___",self.name,self.__spineAni)
    local trackEntity;
    if(type(label) == "number") then
        trackEntity = self.__spineAni.AnimationState:SetAnimationByIndex(0, label, loop);
    else
        trackEntity = self.__spineAni.AnimationState:SetAnimation(0, label, loop);
    end

end

--停止动画
function SpineGraphicExpand:stop()
    self.__spineAni.freeze = true
end


--销毁动画
function SpineGraphicExpand:dispose()
    if (not self.__cobject) then
        return
    end
    self.__spineAni =nil
    self:removeSelf();
    self.__cobject = nil

end

return SpineGraphicExpand
