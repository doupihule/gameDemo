function RandomUtis.hashi_func(num)
  local hash = RandomUtis.crc32(num .. '') >> 16 & 0x7fff
  return hash * 4
end

function RandomUtis.crc32(str, crc)
  if crc == nil then
    crc = 0
  end
  local n = 0
  local x = 0
  crc = crc ^ -1
  local i = 0
  local iTop = #str
  repeat
    n = crc ^ str:charCodeAt(i) & 0xFF
    x = Number("0x" .. RandomUtis.TABLE:substr(n * 9, 8))
    crc = crc >>> 8 ^ x
    i=i+1
  until not(i < iTop)
  return crc ^ -1
end

function RandomUtis.setOneRandomYinzi(yinzi, step, index)
  if step == nil then
    step=0
  end
  if index == nil then
    index=0
  end
  local randomObj = RandomUtis.newYinzi()
  -- [ts2lua]randomObj下标访问可能不正确
  randomObj['initR'] = RandomUtis.hashi_func(yinzi)
  -- [ts2lua]RandomUtis._yinziToIndexDic下标访问可能不正确

  RandomUtis.gotoTargetStep(step, index)
end

function RandomUtis.gotoTargetStep(step, index)
  if index == nil then
    index=0
  end
  -- [ts2lua]RandomUtis._yinziToIndexDic下标访问可能不正确
  -- [ts2lua]RandomUtis._yinziToIndexDic[index]下标访问可能不正确
  -- [ts2lua]RandomUtis._yinziToIndexDic下标访问可能不正确
  -- [ts2lua]RandomUtis._yinziToIndexDic[index]下标访问可能不正确
  RandomUtis._yinziToIndexDic[index]['r'] = RandomUtis._yinziToIndexDic[index]['initR']
  -- [ts2lua]RandomUtis._yinziToIndexDic下标访问可能不正确
  -- [ts2lua]RandomUtis._yinziToIndexDic[index]下标访问可能不正确

function RandomUtis.yinziGetNext(index)
  if index == nil then
    index=0
  end
  local t = RandomUtis.yinziGetOriginUint(index)
  return t % RandomUtis.RANDOM_MAX + 1 / RandomUtis.RANDOM_MAX
end

function RandomUtis.yinziGetOriginUint(index)
  if index == nil then
    index=0
  end
  -- [ts2lua]RandomUtis._yinziToIndexDic下标访问可能不正确
  -- [ts2lua]RandomUtis._yinziToIndexDic[index]下标访问可能不正确
  local quotient = math.floor(RandomUtis._yinziToIndexDic[index]['r'] / RandomUtis.MIDDLEVALUE)
  -- [ts2lua]RandomUtis._yinziToIndexDic下标访问可能不正确
  -- [ts2lua]RandomUtis._yinziToIndexDic[index]下标访问可能不正确
  local remainder = RandomUtis._yinziToIndexDic[index]['r'] % RandomUtis.MIDDLEVALUE + 1
  local t = RandomUtis.MODVALUE * remainder - RandomUtis.OFFSETVALUE * quotient
  if t <= 0 then
    t = t + RandomUtis.RANDOM_MAX
  end

  -- [ts2lua]RandomUtis._yinziToIndexDic[index]下标访问可能不正确

function RandomUtis.getOneRandomFromArea(startNum, endNum, index, damic)
  if index == nil then
    index=0
  end
  if damic == nil then
    damic=0
  end
  if startNum > endNum then
    local tempValue = startNum
    startNum = endNum
    endNum = tempValue
  end
  local random = RandomUtis.getOneRandom(index)
  local k = endNum - startNum / 1 - 0
  local result = k * random - k * 0 + startNum
  -- [ts2lua]lua中0和空字符串也是true，此处result < startNum需要确认
  result = (result < startNum and {startNum} or {result})[1]
  -- [ts2lua]lua中0和空字符串也是true，此处result > endNum需要确认
  result = (result > endNum and {endNum} or {result})[1]
  local powNum = math.pow(10, damic)
  result = math.round(result * powNum) / powNum
  if damic == 0 then
    result = parseInt(result)
  end
  return result
end

function RandomUtis.getOneGroupIndex(restrict, nums, index)
  if index == nil then
    index=0
  end
  if not nums then

  end
  local resultArr = {}
  local tempArr = {}
  local i = 0
  repeat
    tempArr[i+1] = i
    i=i+1
  until not(i < restrict)

  repeat
    local randomInt = RandomUtis.getOneRandomInt(i, 0, index)
    -- [ts2lua]tempArr下标访问可能不正确
    local value_1 = tempArr[randomInt]
    local value_2 = tempArr[i+1]
    table.insert(resultArr, value_1)
    pushIndex = pushIndex + 1
    -- [ts2lua]tempArr下标访问可能不正确
    tempArr[randomInt] = value_2
    tempArr[i+1] = value_1
    i=i-1
  until not(i >= restrict - nums)
  return resultArr
end

function RandomUtis.getOneRandomInt(restrict, min, index)
  if min == nil then
    min=0
  end
  if index == nil then
    index=0
  end
  local nums = restrict - min
  local random = RandomUtis.getOneRandom(index) * nums
  local result = math.floor(random)
  result = result + min
  return result
end

function RandomUtis.getOneIndexByWeight(arr, index)
  if index == nil then
    index=0
  end
  local totalValue = 0
  local tempArr = {}
  local i = 0
  repeat
    totalValue = totalValue + arr[i+1]
    table.insert(tempArr, totalValue)
    i=i+1
  until not(i < #arr)
  local random = RandomUtis.getOneRandom(index)
  random = random * totalValue
  local i = 0
  repeat
    if random < tempArr[i+1] then
      return i
    end
    i=i+1
  until not(i < #tempArr)
  return #tempArr - 1
end

  local randomArr = {}

    LogsManager.errorTag(nil, 'getIndexArrByWeight  ````  arr长度小于nums！！')

function RandomUtis.randomOneGroupToOut(arr, resultArr, index)

    index=0
  end

  local indexArr = RandomUtis.getOneGroupIndex(nums, nums, index)
  if not resultArr then
    resultArr = {}
  end

  repeat
    -- [ts2lua]arr下标访问可能不正确
    resultArr[i+1] = arr[indexArr[i+1]]
    i=i+1
  until not(i < nums)
  return resultArr
