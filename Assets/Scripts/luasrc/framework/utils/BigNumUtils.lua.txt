function BigNumUtils.compareIn(a, b)
  a = a:split("")
  b = b:split("")
  if #a > #b then
    return true
  elseif #a < #b then
    return false
  else
    local i = 0
    repeat
      if tonumber(a[i+1]) > tonumber(b[i+1]) then
        return true
      elseif tonumber(a[i+1]) < tonumber(b[i+1]) then
        return false
      end
      if i == #a - 1 then
        return false
      end
      i=i+1
    until not(i <= #a - 1)
  end
end
