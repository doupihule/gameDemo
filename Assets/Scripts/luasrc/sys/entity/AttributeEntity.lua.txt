AttributeEntity = class('AttributeEntity')

function AttributeEntity:ctor(data)
    local scaleValue = 1
    self.basePower = BattleFunc.getInstance():turnAddSpeedToFrame(data.basePower)
    self.baseSlowDownPower = BattleFunc.getInstance():turnAddSpeedToFrame(data.baseSlowDownPower)
    self.baseMaxSpeed = BattleFunc.getInstance():turnSpeedToFrame(data.baseMaxSpeed)
    self.turnSpeed = BattleFunc.getInstance():turnRotateSpeedToFrame(data.turnSpeed)
    self.turnPower = BattleFunc.getInstance():turnAddSpeedToFrame(data.turnPower)
    self.turnPowerTime = BattleFunc.getInstance():turnMinisecondToframe(data.turnPowerTime)
    self.turnPowerMaxTime = BattleFunc.getInstance():turnMinisecondToframe(data.turnPowerMaxTime)
    self.turnMaxspeed = BattleFunc.getInstance():turnSpeedToFrame(data.turnMaxspeed)
    self.unsafeFallTurnTime = BattleFunc.getInstance():turnMinisecondToframe(data.unsafeFallTurnTime)
    self.unsafeFallSlowDownPower = BattleFunc.getInstance():turnAddSpeedToFrame(data.unsafeFallSlowDownPower)
    self.slowDownPower = BattleFunc.getInstance():turnAddSpeedToFrame(data.slowDownPower)
    self.slowDownPower = BattleFunc.getInstance():turnAddSpeedToFrame(data.slowDownPower)
    self.collisionSpeedChange = TableUtils.copyOneArr(data.collisionSpeedChange)
    self.collisionSpeedChange[0 + 1] = self.collisionSpeedChange[0 + 1] / 10000
    self.collisionSpeedChange[1 + 1] = self.collisionSpeedChange[1 + 1] / 10000
    self.roadChangeDirection = data.roadChangeDirection
    self.skyChangeDirection = data.skyChangeDirection
    self.carLength = data.carLength / 10000
    self.box = data.box
end
