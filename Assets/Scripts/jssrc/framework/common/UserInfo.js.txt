        var obj = {};
        if (this.paramStr) {
            var params = this.paramStr.split('&');
            var len = params.length;
            for (var i = 0; i < len; i++) {
                var value = params[i].replace(/#.*$/g, '').split('=');
                if (!value[1]) {
                    value[1] = '';

                obj[value[0]] = value[1];
